----
import wixData from 'wix-data';

$w.onReady(function () {
    $w('#button4').onClick(() => {
        search();
    })

});

function search() {

    wixData.query("Courses1")
        .eq("subjectName", $w('#button4').label)
        .find()
        .then(results => {
            $w('#listRepeater').data = results.items;
        });

}
